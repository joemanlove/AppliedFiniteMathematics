\section{Logarithm Puzzles}

\begin{puzzle}
    For each equation, rewrite in exponential form and solve for \(x\):
    \begin{enumerate}
        \item \(\log_5 (x) = 3\)
        \item \(\log_2 (x) = -2\)
        \item \(\log_{10} (x) = -3\)
        \item \(\log_3 (x) = 6\)
        \item \(\log_{25} (x) = \frac{1}{2}\)
        \item \(\log_{64} (x) = \frac{1}{3}\)
    \end{enumerate}
\end{puzzle}

\begin{puzzle}
    Evaluate without using your calculator:
    \begin{enumerate}
        \item \(\ln (\sqrt[3]{e})\)
        \item \(\ln \left(\frac{1}{e^2}\right)\)
        \item \(\ln (e^{10})\)
        \item \(\log_{10} (10^{e})\)
    \end{enumerate}
\end{puzzle}

\begin{puzzle}
    Evaluate the following logarithms using a calculator or computer.
    \begin{enumerate}
        \item \(\log_{10} (20)\)
        \item \(\ln (42)\)
        \item \(\ln (2.9)\)
        \item \(\log_{10} (0.5)\)
        \item \(\log_{4} (36)\)
        \item \(\log_{7} (100)\)
        \item \(\log_{10.5} (3.5)\)
        \item \(\log_{1.067} (2)\)
    \end{enumerate}
\end{puzzle}
